📁 index.js
js
Copy
Edit
const express = require('express');
const app = express();
const PORT = process.env.PORT || 3000;

app.set('view engine', 'ejs');
app.use(express.static('public'));

// Route: Home Dashboard
app.get('/', (req, res) => {
  res.render('dashboard');
});

// Placeholder routes
app.get('/internet', (req, res) => res.send('Buy Internet Page Coming Soon'));
app.get('/boda', (req, res) => res.send('BodaBoda Page Coming Soon'));
app.get('/restaurants', (req, res) => res.send('Restaurant Order Page Coming Soon'));
app.get('/shop', (req, res) => res.send('Local Shop Page Coming Soon'));
app.get('/jobs', (req, res) => res.send('Local Jobs Page Coming Soon'));

app.listen(PORT, () => {
  console.log(`🚀 SuperApp dashboard running on port ${PORT}`);
});
